    void selection_sort(int array[], int n) {

### 3. æœç´¢ (Search)

#### A. Linear Search (ç·šæ€§æœç´¢)
* **æ™‚é–“è¤‡é›œåº¦ (Time Complexity):** $O(n)$
* æ•ˆç‡æœ€å·®çš„æ–¹å¼ï¼Œå› ç‚ºå¾—å…¨éƒ¨æ‰¾éï¼Œæ‰èƒ½ç¢ºèªåœ¨å“ªã€‚

    ```c
    #include <stdio.h>

    int find(int arr[], int arr_size, int target){
        // The worst case: you should find the last element in this array
        // The time complexity is O(n)
        for(int i = 0; i < arr_size; i++){
            if(arr[i] == target) return i;
        }
        return -1; // Added: Return -1 if not found
    }

    int main()
    {
        int arr[5] = {1, 2, 3, 4, 5};
        int arr_size = 5;
        int target = 4;
        printf("Target index: %d\n", find(arr, arr_size, target));
        return 0;
    }
    ```

#### B. Binary Search (äºŒåˆ†æœç´¢æ³•)
* **æ™‚é–“è¤‡é›œåº¦ (Time Complexity):** $O(\log N)$

    ```c
    #include <stdio.h>

    int binary_search(int arr[], int arr_size, int target){
        int Left = 0, Right = arr_size - 1, current;

        while(Left <= Right){
            current = Left + (Right - Left) / 2; // Better way to calculate middle index

            if(arr[current] > target){
                Right = current - 1;
            }
            else if(arr[current] < target){
                Left  = current + 1;
            }
            else{
                return current; // Found
            }
        }
        return -1; // Added: Return -1 if not found
    }

    int main()
    {
        int arr[5] = {1, 2, 3, 4, 5}; // Note: Binary Search requires a sorted array
        int arr_size = 5;
        int target = 4;
        printf("Target index: %d\n", binary_search(arr, arr_size, target));
        return 0;
    }
    ```
